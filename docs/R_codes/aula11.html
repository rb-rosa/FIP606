library(readxl)
library(tidyverse)
dat <- read_excel("dados-diversos.xlsx", "fungicida_vaso")

dat2 <- dat |> 
  # mutate para criar uma variável
  mutate(inc = dis_sp / n_sp*100)

dat2 |>
  ggplot(aes(x = treat,
             y = inc))+
  geom_jitter(width = 0.1)+
  facet_wrap(~ dose)

Checar premissas, comparar médias, CV
